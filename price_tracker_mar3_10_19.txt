#!/usr/bin/env python
# manual_parser.py

import sys
import time
from playwright.sync_api import sync_playwright
from datetime import datetime
import os
import django
import traceback

# Настройка Django
sys.path.append('/home/artem/work/price_tracker')
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'config.settings')
django.setup()

from tracker.models import Product


def manual_parse(product_id):
    """Ручной парсинг товара с открытием браузера"""
    print("\n" + "=" * 60)
    print(f"🔍 РУЧНОЙ ПАРСИНГ ТОВАРА #{product_id}")
    print("=" * 60)

    try:
        product = Product.objects.get(id=product_id)
        print(f"📦 Название: {product.name or 'Без названия'}")
        print(f"🔗 URL: {product.url}")
        print("=" * 60 + "\n")
    except Product.DoesNotExist:
        print(f"❌ Товар с ID {product_id} не найден")
        return

    browser = None
    try:
        with sync_playwright() as p:
            # Запускаем браузер видимо
            browser = p.chromium.launch(
                headless=False,
                args=['--disable-blink-features=AutomationControlled']
            )

            context = browser.new_context(
                viewport={'width': 1280, 'height': 1024},
                locale='ru-RU',
                user_agent='Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'
            )

            page = context.new_page()

            print("🌐 Открываю страницу...")
            page.goto(product.url, timeout=60000)

            print("\n✅ Страница загружена. Браузер остаётся открытым.")
            print("📢 Если появилась капча - решите её вручную.")
            print("🔄 После решения капчи нажмите Enter в этом окне...")
            input()

            # Даем время на загрузку
            page.wait_for_load_state('networkidle')
            time.sleep(2)

            # Получаем данные
            title = page.title()
            print(f"\n📑 Заголовок: {title}")

            # Парсим название (h1)
            name = page.query_selector('h1')
            product_name = name.text_content().strip() if name else title.split('—')[0].strip()
            print(f"📝 Название: {product_name}")

            # Парсим цену
            price_selectors = [
                'span[itemprop="price"]',
                'span[class*="price"]',
                'span[class*="tsHeadline500Medium"]',
                'div[data-widget="webPrice"] span',
                'span[content]'
            ]

            price = None
            for selector in price_selectors:
                price_elem = page.query_selector(selector)
                if price_elem:
                    price_text = price_elem.text_content()
                    if price_text:
                        import re
                        numbers = re.findall(r'(\d+[\.,]?\d*)', price_text.replace(' ', ''))
                        if numbers:
                            try:
                                potential_price = float(numbers[0].replace(',', '.'))
                                if potential_price > 10000:
                                    potential_price = potential_price / 100
                                price = potential_price
                                print(f"💰 Найдена цена: {price} ₽")
                                break
                            except:
                                continue

            if not price:
                print("⚠️ Цена не найдена")

            # Парсим изображение
            img_selectors = [
                'meta[property="og:image"]',
                'img[itemprop="image"]',
                'div[data-widget="webGallery"] img'
            ]

            image_url = None
            for selector in img_selectors:
                if selector.startswith('meta'):
                    img = page.query_selector(selector)
                    if img:
                        image_url = img.get_attribute('content')
                else:
                    img = page.query_selector(selector)
                    if img:
                        image_url = img.get_attribute('src')
                if image_url:
                    if image_url.startswith('//'):
                        image_url = 'https:' + image_url
                    print(f"🖼️ Изображение найдено")
                    break

            # Делаем скриншот
            screenshot_dir = os.path.expanduser("~/price_tracker_data/screenshots")
            os.makedirs(screenshot_dir, exist_ok=True)

            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            screenshot_path = os.path.join(screenshot_dir, f"product_{product_id}_{timestamp}.png")
            page.screenshot(path=screenshot_path, full_page=True)
            print(f"📸 Скриншот сохранен: {screenshot_path}")

            # Сохраняем в базу
            product.name = product_name[:500]
            product.current_price = price
            product.image_url = image_url
            product.save()

            print(f"\n✅ Данные сохранены в базу!")
            print(f"   Название: {product_name}")
            print(f"   Цена: {price} ₽" if price else "   Цена не найдена")

    except Exception as e:
        print(f"\n❌ Ошибка: {e}")
        traceback.print_exc()
    finally:
        if browser:
            print("\n🔄 Закрываю браузер...")
            browser.close()

        input("\n✅ Нажмите Enter для закрытия окна...")


if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Использование: python manual_parser.py <product_id>")
        sys.exit(1)

    try:
        product_id = int(sys.argv[1])
        manual_parse(product_id)
    except ValueError:
        print("❌ Неверный ID товара")
        sys.exit(1)

def parse_ozon_manually(product_id):
    """
    Ручной парсинг Ozon с видимым браузером
    """
    try:
        product = Product.objects.get(id=product_id)
    except Product.DoesNotExist:
        print(f"❌ Товар с ID {product_id} не найден")
        return

    print("\n" + "=" * 60)
    print(f"🔍 РУЧНОЙ ПАРСИНГ OZON")
    print(f"📦 Товар ID: {product.id}")
    print(f"🔗 URL: {product.url}")
    print("=" * 60 + "\n")

    with sync_playwright() as p:
        # Запускаем браузер видимо
        browser = p.chromium.launch(
            headless=False,
            args=['--disable-blink-features=AutomationControlled']
        )

        context = browser.new_context(
            viewport={'width': 1280, 'height': 1024},
            locale='ru-RU',
            user_agent='Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'
        )

        page = context.new_page()

        print("🌐 Открываю страницу...")
        page.goto(product.url)

        # Проверяем капчу
        if "captcha" in page.content().lower():
            print("\n🚫 ОБНАРУЖЕНА КАПЧА!")
            input("👉 Решите капчу в браузере, затем нажмите Enter...")

        # Ждем загрузки
        page.wait_for_load_state('networkidle')
        time.sleep(2)

        # Получаем данные
        title = page.title()
        print(f"\n📑 Заголовок: {title}")

        # Название - ищем самый большой заголовок
        name_selectors = ['h1', '.product-title', '[data-widget="webProductHeading"] h1']
        product_name = None
        for selector in name_selectors:
            name_elem = page.query_selector(selector)
            if name_elem:
                product_name = name_elem.text_content().strip()
                break

        if product_name:
            print(f"📝 Название: {product_name}")
            product.name = product_name
        else:
            # Если не нашли, используем часть заголовка
            product_name = title.split('купить')[0].strip()
            if product_name:
                product.name = product_name
                print(f"📝 Название (из заголовка): {product_name}")

        # Цена - ищем крупные числа
        price_selectors = [
            'span[itemprop="price"]',
            'span[class*="price"]',
            'span[class*="tsHeadline500Medium"]',
            'div[data-widget="webPrice"] span',
            'span[content]'
        ]

        price = None
        for selector in price_selectors:
            price_elem = page.query_selector(selector)
            if price_elem:
                price_text = price_elem.text_content()
                if price_text:
                    import re
                    # Ищем числа больше 100 (чтобы отсечь мелкие цифры)
                    numbers = re.findall(r'(\d+[\.,]?\d*)', price_text.replace(' ', ''))
                    for num in numbers:
                        try:
                            potential_price = float(num.replace(',', '.'))
                            # Настоящая цена обычно > 100
                            if potential_price > 100:
                                if potential_price > 10000:
                                    potential_price = potential_price / 100
                                price = potential_price
                                break
                        except:
                            continue
                if price:
                    break

        if price:
            print(f"💰 Цена: {price} ₽")
            product.current_price = price
        else:
            print("⚠️ Цена не найдена")

        # Изображение
        img_selectors = [
            'meta[property="og:image"]',
            'img[itemprop="image"]',
            'div[data-widget="webGallery"] img'
        ]

        for selector in img_selectors:
            if selector.startswith('meta'):
                img = page.query_selector(selector)
                if img:
                    img_url = img.get_attribute('content')
            else:
                img = page.query_selector(selector)
                if img:
                    img_url = img.get_attribute('src')
            if img_url:
                if img_url.startswith('//'):
                    img_url = 'https:' + img_url
                product.image_url = img_url
                print(f"🖼️ Изображение: {img_url[:100]}...")
                break

        # Сохраняем
        product.save()
        print(f"\n✅ Данные сохранены в базу")

        # Делаем скриншот
        timestamp = int(time.time())
        screenshot_file = f"ozon_manual_{product.id}_{timestamp}.png"
        page.screenshot(path=screenshot_file)
        print(f"📸 Скриншот сохранен в {screenshot_file}")

        input("\nНажмите Enter для закрытия браузера...")
        browser.close()

from django.db import models
from django.contrib.auth.models import User
from django.utils import timezone


class Product(models.Model):
    PLATFORM_CHOICES = [
        ('wb', 'Wildberries'),
        ('ozon', 'Ozon'),
        ('ali', 'AliExpress'),
    ]

    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='products')
    url = models.URLField(max_length=500)
    platform = models.CharField(max_length=10, choices=PLATFORM_CHOICES)
    name = models.CharField(max_length=500)
    current_price = models.DecimalField(max_digits=10, decimal_places=2, null=True, blank=True)
    desired_price = models.DecimalField(max_digits=10, decimal_places=2)
    image_url = models.URLField(max_length=500, blank=True, null=True)
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    last_checked = models.DateTimeField(null=True, blank=True)

    class Meta:
        ordering = ['-created_at']

    def __str__(self):
        return f"{self.name} - {self.current_price}₽"

    def save(self, *args, **kwargs):
        if not self.platform:
            if 'wildberries' in self.url or 'wb.ru' in self.url:
                self.platform = 'wb'
            elif 'ozon' in self.url:
                self.platform = 'ozon'
            elif 'aliexpress' in self.url:
                self.platform = 'ali'
        super().save(*args, **kwargs)

import logging

@shared_task
def parse_product_price(product_id):
    """
    Заглушка - парсинг теперь происходит в отдельном скрипте
    """
    logger.info(f"📝 Задача на парсинг товара {product_id} создана")
    return f"Запущен ручной парсинг товара {product_id}"


@shared_task
def create_notification(product_id, price):
    """
    Создание уведомления о достижении цены
    """
    try:
        product = Product.objects.get(id=product_id)

        # Проверяем, не отправляли ли уже уведомление сегодня
        recent_notification = Notification.objects.filter(
            product=product,
            is_sent=True,
            created_at__date=timezone.now().date()
        ).exists()

        if recent_notification:
            logger.info(f"⏭️ Уведомление уже отправлено сегодня")
            return f"Уведомление уже отправлено"

        message = f"🎉 Товар '{product.name}' достиг желаемой цены! Текущая цена: {price}₽"

        notification = Notification.objects.create(
            user=product.user,
            product=product,
            message=message
        )
        logger.info(f"📬 Уведомление создано")

        # Отправляем email
        send_notification_email.delay(notification.id)

        return f"Уведомление создано"
    except Exception as e:
        logger.error(f"Ошибка создания уведомления: {e}")
        return f"Ошибка: {e}"


@shared_task
def send_notification_email(notification_id):
    """
    Отправка email уведомления
    """
    try:
        notification = Notification.objects.get(id=notification_id, is_sent=False)

        subject = f"💰 Цена достигла желаемого уровня! - {notification.product.name}"
        message = f"""
        Здравствуйте, {notification.user.username}!

        {notification.message}

        Товар: {notification.product.name}
        Текущая цена: {notification.product.current_price}₽
        Желаемая цена: {notification.product.desired_price}₽
        Ссылка на товар: {notification.product.url}

        --
        Price Tracker
        """

        # Для разработки - вывод в консоль
        logger.info(f"📧 Email для {notification.user.email}:")
        logger.info(f"Тема: {subject}")
        logger.info(f"Сообщение: {message}")

        notification.is_sent = True
        notification.sent_at = timezone.now()
        notification.save()

        return f"Email готов к отправке"
    except Exception as e:
        logger.error(f"Ошибка отправки email: {e}")
        return f"Ошибка: {e}"

    const productId = {{ product.id }};
    const ws = new WebSocket('ws://localhost:8765');

    ws.onopen = function() {
        document.getElementById('status').innerHTML = '✅ Подключено к браузеру. Загружаем страницу...';
        ws.send(JSON.stringify({
            command: 'start_browser',
            url: '{{ product.url }}'
        }));
    };

    ws.onmessage = function(event) {
        const data = JSON.parse(event.data);

        if (data.status === 'browser_started') {
            document.getElementById('status').innerHTML = '✅ Браузер запущен. Загружаем страницу...';

        } else if (data.status === 'waiting_captcha') {
            document.getElementById('status').innerHTML = '🔄 ' + data.message;
            document.getElementById('solve-captcha').style.display = 'inline-block';

            // Запрашиваем скриншот каждые 2 секунды
            setInterval(function() {
                ws.send(JSON.stringify({command: 'get_screenshot'}));
            }, 2000);

        } else if (data.status === 'screenshot') {
            // Обновляем изображение
            const img = document.getElementById('browser-screen');
            img.src = 'data:image/png;base64,' + data.data;
            img.style.display = 'block';

        this.disabled = true;

# tracker/views.py
from django.shortcuts import render, redirect, get_object_or_404
from django.contrib.auth.decorators import login_required
from django.contrib.auth import login
from django.contrib import messages
from django.http import JsonResponse
from .models import Product, PriceHistory, Notification
from .forms import CustomUserCreationForm, ProductForm
import json
import logging
import subprocess
import os

logger = logging.getLogger(__name__)


def index(request):
    """Главная страница"""
    return render(request, 'tracker/index.html')


def register(request):
    """Регистрация нового пользователя"""
    if request.method == 'POST':
        form = CustomUserCreationForm(request.POST)
        if form.is_valid():
            user = form.save()
            login(request, user)
            messages.success(request, '🎉 Регистрация прошла успешно!')
            return redirect('profile')
        else:
            messages.error(request, 'Пожалуйста, исправьте ошибки в форме')
    else:
        form = CustomUserCreationForm()
    return render(request, 'registration/register.html', {'form': form})


@login_required
def profile(request):
    """Профиль пользователя со списком товаров"""
    products = request.user.products.all()
    notifications = request.user.notifications.filter(is_sent=False)[:10]
    return render(request, 'tracker/profile.html', {
        'products': products,
        'notifications': notifications
    })


@login_required
def add_product(request):
    """Добавление нового товара"""
    if request.method == 'POST':
        form = ProductForm(request.POST)
        if form.is_valid():
            product = form.save(commit=False)
            product.user = request.user
            product.save()

            messages.success(request, '✅ Товар успешно добавлен!')
            return redirect('profile')
    else:
        form = ProductForm()

    return render(request, 'tracker/add_product.html', {'form': form})

@login_required
def product_detail(request, product_id):
    """Детальная страница товара с графиком цен"""
    product = get_object_or_404(Product, id=product_id, user=request.user)
    price_history = product.price_history.all()[:50]

    dates = [h.created_at.strftime('%d.%m %H:%M') for h in price_history]
    prices = [float(h.price) for h in price_history]

    context = {
        'product': product,
        'price_history': price_history,
        'dates_json': json.dumps(dates[::-1]),
        'prices_json': json.dumps(prices[::-1]),
    }
    return render(request, 'tracker/product_detail.html', context)


@login_required
def delete_product(request, product_id):
    """Удаление товара"""
    product = get_object_or_404(Product, id=product_id, user=request.user)

    if request.method == 'POST':
        product_name = product.name
        product.delete()
        messages.success(request, f'🗑️ Товар "{product_name}" успешно удален')
    else:
        messages.error(request, '❌ Неверный метод запроса')

    return redirect('profile')


@login_required
def manual_parse(request, product_id):
    """Запуск ручного парсинга в отдельном окне"""
    product = get_object_or_404(Product, id=product_id, user=request.user)

    # Путь к скрипту
    script_path = os.path.join(os.path.dirname(__file__), '..', 'manual_parser.py')

    if not os.path.exists(script_path):
        messages.error(request, '❌ Скрипт парсинга не найден')
        return redirect('profile')

    try:
        subprocess.Popen([
            'xterm', '-e',
            f'python "{script_path}" {product.id}; echo; echo "Нажмите Enter для закрытия..."; read'
        ])
        messages.success(request,
                         f'🖥️ Открыто окно для ручного парсинга товара "{product.name}". Решите капчу в браузере.')
    except FileNotFoundError:
        try:
            subprocess.Popen([
                'gnome-terminal', '--', 'bash', '-c',
                f'python "{script_path}" {product.id}; echo; echo "Нажмите Enter для закрытия..."; read'
            ])
            messages.success(request,
                             f'🖥️ Открыто окно для ручного парсинга товара "{product.name}". Решите капчу в браузере.')
        except:
            messages.error(request, '❌ Не удалось открыть терминал. Установите xterm или gnome-terminal.')

    return redirect('profile')
