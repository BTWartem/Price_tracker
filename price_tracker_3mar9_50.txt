import os
import sys


def main():
    """Run administrative tasks."""
    os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'config.settings')
    try:
        from django.core.management import execute_from_command_line

    main()

#!/usr/bin/env python
# manual_ozon_parser.py

import sys
import os
import django
import time
from pathlib import Path

# Настройка Django
sys.path.append(str(Path(__file__).parent))
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'config.settings')
django.setup()

from playwright.sync_api import sync_playwright
from tracker.models import Product
from tracker.tasks import parse_product_price


def parse_ozon_manually(product_id):
    """
    Ручной парсинг Ozon с видимым браузером
    """
    try:
        product = Product.objects.get(id=product_id)
    except Product.DoesNotExist:
        print(f"❌ Товар с ID {product_id} не найден")
        return

    print("\n" + "=" * 60)
    print(f"🔍 РУЧНОЙ ПАРСИНГ OZON")
    print(f"📦 Товар ID: {product.id}")
    print(f"🔗 URL: {product.url}")
    print("=" * 60 + "\n")

    with sync_playwright() as p:
        # Запускаем браузер видимо
        browser = p.chromium.launch(
            headless=False,
            args=['--disable-blink-features=AutomationControlled']
        )

        context = browser.new_context(
            viewport={'width': 1280, 'height': 1024},
            locale='ru-RU',
            user_agent='Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'
        )

        page = context.new_page()

        print("🌐 Открываю страницу...")
        page.goto(product.url)

        # Проверяем капчу
        if "captcha" in page.content().lower():
            print("\n🚫 ОБНАРУЖЕНА КАПЧА!")
            input("👉 Решите капчу в браузере, затем нажмите Enter...")

        # Ждем загрузки
        page.wait_for_load_state('networkidle')
        time.sleep(2)

        # Получаем данные
        title = page.title()
        print(f"\n📑 Заголовок: {title}")

        # Название - ищем самый большой заголовок
        name_selectors = ['h1', '.product-title', '[data-widget="webProductHeading"] h1']
        product_name = None
        for selector in name_selectors:
            name_elem = page.query_selector(selector)
            if name_elem:
                product_name = name_elem.text_content().strip()
                break

        if product_name:
            print(f"📝 Название: {product_name}")
            product.name = product_name
        else:
            # Если не нашли, используем часть заголовка
            product_name = title.split('купить')[0].strip()
            if product_name:
                product.name = product_name
                print(f"📝 Название (из заголовка): {product_name}")

        # Цена - ищем крупные числа
        price_selectors = [
            'span[itemprop="price"]',
            'span[class*="price"]',
            'span[class*="tsHeadline500Medium"]',
            'div[data-widget="webPrice"] span',
            'span[content]'
        ]

        price = None
        for selector in price_selectors:
            price_elem = page.query_selector(selector)
            if price_elem:
                price_text = price_elem.text_content()
                if price_text:
                    import re
                    # Ищем числа больше 100 (чтобы отсечь мелкие цифры)
                    numbers = re.findall(r'(\d+[\.,]?\d*)', price_text.replace(' ', ''))
                    for num in numbers:
                        try:
                            potential_price = float(num.replace(',', '.'))
                            # Настоящая цена обычно > 100
                            if potential_price > 100:
                                if potential_price > 10000:
                                    potential_price = potential_price / 100
                                price = potential_price
                                break
                        except:
                            continue
                if price:
                    break

        if price:
            print(f"💰 Цена: {price} ₽")
            product.current_price = price
        else:
            print("⚠️ Цена не найдена")

        # Изображение
        img_selectors = [
            'meta[property="og:image"]',
            'img[itemprop="image"]',
            'div[data-widget="webGallery"] img'
        ]

        for selector in img_selectors:
            if selector.startswith('meta'):
                img = page.query_selector(selector)
                if img:
                    img_url = img.get_attribute('content')
            else:
                img = page.query_selector(selector)
                if img:
                    img_url = img.get_attribute('src')
            if img_url:
                if img_url.startswith('//'):
                    img_url = 'https:' + img_url
                product.image_url = img_url
                print(f"🖼️ Изображение: {img_url[:100]}...")
                break

        # Сохраняем
        product.save()
        print(f"\n✅ Данные сохранены в базу")

        # Делаем скриншот
        timestamp = int(time.time())
        screenshot_file = f"ozon_manual_{product.id}_{timestamp}.png"
        page.screenshot(path=screenshot_file)
        print(f"📸 Скриншот сохранен в {screenshot_file}")

        input("\nНажмите Enter для закрытия браузера...")
        browser.close()


def main():
    """Основная функция"""
    if len(sys.argv) > 1:
        # Если передан ID как аргумент
        try:
            product_id = int(sys.argv[1])
            parse_ozon_manually(product_id)
        except ValueError:
            print("❌ Неверный ID товара")
    else:
        # Интерактивный режим
        print("\n🔧 РУЧНОЙ ПАРСИНГ OZON")
        print("=" * 60)

        # Показываем список товаров
        products = Product.objects.all()
        print("\n📋 Доступные товары:")
        for p in products:
            status = "✅" if p.current_price else "⏳"
            print(f"  {status} ID: {p.id} - {p.name or 'Без названия'} - {p.url[:50]}...")

        # Запрашиваем ID
        try:
            product_id = input("\n🔢 Введите ID товара для парсинга (или 'q' для выхода): ")
            if product_id.lower() == 'q':
                return
            parse_ozon_manually(int(product_id))
        except ValueError:
            print("❌ Неверный ID")
        except KeyboardInterrupt:
            print("\n👋 Выход")


if __name__ == "__main__":
    main()

import asyncio
import websockets
import json
import threading
from playwright.async_api import async_playwright
import logging

logger = logging.getLogger(__name__)


class BrowserServer:
    """Сервер для удаленного управления браузером"""

    def __init__(self, port=8765):
        self.port = port
        self.browser = None
        self.page = None
        self.captcha_solved = asyncio.Event()
        self.html_content = None

    async def start(self):
        """Запускает WebSocket сервер"""
        async with websockets.serve(self.handle_connection, "localhost", self.port):
            logger.info(f"✅ Browser server started on ws://localhost:{self.port}")
            await asyncio.Future()  # бесконечное ожидание

    async def handle_connection(self, websocket):
        """Обрабатывает WebSocket соединение"""
        try:
            async for message in websocket:
                data = json.loads(message)
                cmd = data.get('command')

                if cmd == 'start_browser':
                    # Запускаем браузер
                    await self.start_browser(data.get('url'))
                    await websocket.send(json.dumps({'status': 'browser_started'}))

                elif cmd == 'page_ready':
                    # Страница загружена, ждем решения капчи
                    await websocket.send(json.dumps({
                        'status': 'waiting_captcha',
                        'message': 'Решите капчу в открытом окне браузера'
                    }))

                elif cmd == 'captcha_solved':
                    # Капча решена
                    self.captcha_solved.set()
                    # Получаем HTML
                    self.html_content = await self.page.content()
                    await websocket.send(json.dumps({
                        'status': 'success',
                        'html': self.html_content
                    }))

                elif cmd == 'get_screenshot':
                    # Отправляем скриншот
                    screenshot = await self.page.screenshot()
                    await websocket.send(json.dumps({
                        'status': 'screenshot',
                        'data': screenshot.hex()
                    }))

        except Exception as e:
            logger.error(f"WebSocket error: {e}")

    async def start_browser(self, url):
        """Запускает браузер с видимым окном"""
        p = await async_playwright().start()
        self.browser = await p.chromium.launch(
            headless=False,
            args=['--disable-blink-features=AutomationControlled']
        )
        context = await self.browser.new_context(
            viewport={'width': 1280, 'height': 1024},
            locale='ru-RU'
        )
        self.page = await context.new_page()
        await self.page.goto(url)

    async def close(self):
        """Закрывает браузер"""
        if self.browser:
            await self.browser.close()


# Глобальный экземпляр сервера
browser_server = BrowserServer()


def run_server():
    """Запускает сервер в отдельном потоке"""
    asyncio.run(browser_server.start())


# Запускаем сервер при импорте
thread = threading.Thread(target=run_server, daemon=True)
thread.start()

PROJECT_DIR = os.path.dirname(os.path.dirname(os.path.dirname(__file__)))


class ScrapyRunner:
    def __init__(self, url):
        self.url = url
        self.spider_name = self._get_spider_name()

    def _get_spider_name(self):
        url_lower = self.url.lower()
        if 'ozon' in url_lower:
            return 'ozon'
        elif 'wildberries' in url_lower:
            return 'wildberries'
        elif 'aliexpress' in url_lower:
            return 'aliexpress'
        else:
            return 'universal'

    def _create_universal_spider(self):
        spider_code = '''import scrapy
class UniversalSpider(scrapy.Spider):
    name = "universal"
    def __init__(self, url=None):
        self.start_urls = [url] if url else []
    def parse(self, response):
        yield {'name': response.css('title::text').get() or 'Нет названия', 'price': 0, 'url': response.url}
'''
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(spider_code)
            return f.name

    def parse(self):
        if not self.spider_name:
            logger.error(f"No spider for URL: {self.url}")
            return None

        logger.info(f"🕷️ Запускаем паука {self.spider_name} для: {self.url}")

        with tempfile.NamedTemporaryFile(mode='w+', suffix='.json', delete=False) as tmp:
            output_file = tmp.name

        try:
            if self.spider_name == 'universal':
                spider_file = self._create_universal_spider()
                cmd = [SCRAPY_PATH, 'runspider', spider_file, '-a', f'url={self.url}', '-o', f'{output_file}:json',
                       '--nolog']
            else:
                # Для обычных пауков
                cmd = [
                    SCRAPY_PATH, 'crawl', self.spider_name,
                    '-a', f'url={self.url}',
                    '-o', f'{output_file}:json',
                    '--nolog'
                ]

            env = os.environ.copy()
            env['PYTHONPATH'] = f"{PROJECT_DIR}:{env.get('PYTHONPATH', '')}"
            env['SCRAPY_SETTINGS_MODULE'] = 'tracker.parsers.scrapy_spiders.settings'

            logger.info(f"📋 Команда: {' '.join(cmd)}")

            # Запускаем процесс
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=60,  # Увеличиваем таймаут для ручного режима
                env=env,
                cwd=PROJECT_DIR
            )

            if result.returncode != 0:
                logger.error(f"❌ Scrapy ошибка (код {result.returncode}):")
                if result.stderr:
                    # Фильтруем предупреждения
                    errors = [line for line in result.stderr.split('\n')
                              if 'UserWarning' not in line and 'DeprecationWarning' not in line]
                    if errors:
                        logger.error(f"STDERR: {chr(10).join(errors)}")
                return None

            # Читаем результаты
            if os.path.exists(output_file):
                with open(output_file, 'r') as f:
                    content = f.read().strip()
                    if content:
                        try:
                            data = json.loads(content)
                            if data and len(data) > 0:
                                item = data[0]
                                logger.info(f"✅ Scrapy успешно получил данные")
                                return {
                                    'name': item.get('name', ''),
                                    'price': float(item.get('price', 0)) if item.get('price') else 0,
                                    'image_url': item.get('image_url'),
                                    'platform': item.get('platform', 'unknown')
                                }
                        except json.JSONDecodeError as e:
                            logger.error(f"❌ Ошибка парсинга JSON: {e}")
            else:
                logger.error(f"❌ Файл результатов не создан: {output_file}")

        except subprocess.TimeoutExpired:
            logger.error("⏰ Scrapy process timeout (60 сек)")
        except Exception as e:
            logger.error(f"❌ Неожиданная ошибка: {e}")
        finally:
            if os.path.exists(output_file):
                os.unlink(output_file)

        return None

import scrapy

class AliExpressSpider(scrapy.Spider):
    name = "aliexpress"
    allowed_domains = ["aliexpress.ru", "aliexpress.com"]

    custom_settings = {
        'DOWNLOAD_DELAY': 2.0,
        'CONCURRENT_REQUESTS': 1,
        'RETRY_TIMES': 3,
        'USER_AGENT': 'Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
    }

    def __init__(self, url=None, *args, **kwargs):
        super().__init__(*args, **kwargs)
        if url:
            self.start_urls = [url]

    def parse(self, response):
        self.logger.info(f"✅ Страница AliExpress загружена")

        item = ProductItem()

        # 1. Поиск названия
        name_selectors = [
            'h1::text',
            'h1[class*="product-title"]::text',
            'meta[property="og:title"]::attr(content)',
        ]

        for selector in name_selectors:
            if selector.startswith('meta'):
                name = response.css(selector).get()
            else:
                name = response.css(selector).get()
            if name and name.strip():
                item['name'] = name.strip()
                # Чистим от лишнего
                item['name'] = re.sub(r' - AliExpress.*$', '', item['name'])
                self.logger.info(f"📝 Название: {item['name'][:50]}...")
                break

        # 2. Поиск цены
        price_selectors = [
            'span[itemprop="price"]::attr(content)',
            'span[class*="price"]::text',
            'div[class*="price"]::text',
            'span[class*="snow-price"]::text',
        ]

        for selector in price_selectors:
            price_text = response.css(selector).get()
            if price_text:
                numbers = re.findall(r'(\d+[\.,]?\d*)', price_text.replace(' ', ''))
                if numbers:
                    try:
                        price = float(numbers[0].replace(',', '.'))
                        if price > 10000:
                            price = price / 100
                        item['price'] = price
                        self.logger.info(f"💰 Цена: {price}")
                        break
                    except:
                        continue

        # 3. Поиск изображения
        img_selectors = [
            'meta[property="og:image"]::attr(content)',
            'img[class*="product-image"]::attr(src)',
        ]

        for selector in img_selectors:
            if selector.startswith('meta'):
                img_url = response.css(selector).get()
            else:
                img_url = response.css(selector).get()
            if img_url:
                if img_url.startswith('//'):
                    img_url = 'https:' + img_url
                item['image_url'] = img_url
                break

        item['url'] = response.url
        item['platform'] = 'aliexpress'

        if item.get('name') or item.get('price'):
            yield item

# tracker/parsers/scrapy_spiders/spiders/ozon_spider.py
import scrapy
import asyncio
import websockets
import json
from ..items import ProductItem


class OzonSpider(scrapy.Spider):
    name = "ozon"
    allowed_domains = ["ozon.ru", "www.ozon.ru"]

    def __init__(self, url=None, product_id=None, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.url = url
        self.product_id = product_id
        self.start_urls = [url] if url else []

    def start_requests(self):
        for url in self.start_urls:
            yield scrapy.Request(
                url,
                meta={
                    'product_id': self.product_id
                },
                callback=self.parse,
                errback=self.errback
            )

    async def parse(self, response):
        # Проверяем на капчу
        if "captcha" in response.text.lower() or "проверка" in response.text.lower():
            self.logger.warning(f"🚫 Обнаружена капча для товара {self.product_id}")

            # Отправляем уведомление через WebSocket
            async with websockets.connect('ws://localhost:8765') as websocket:
                await websocket.send(json.dumps({
                    'command': 'captcha_detected',
                    'product_id': self.product_id,
                    'url': self.url
                }))

                # Ждем решения капчи
                response = await websocket.recv()
                data = json.loads(response)

                if data.get('html'):
                    # Создаем новый response с полученным HTML
                    from scrapy.http import HtmlResponse
                    response = HtmlResponse(
                        url=self.url,
                        body=data['html'],
                        encoding='utf-8',
                        request=response.request
                    )
                    self.logger.info(f"✅ Капча решена, продолжаем парсинг")
                else:
                    self.logger.error(f"❌ Не удалось решить капчу")
                    return

        # Парсим страницу
        item = ProductItem()

        # Поиск названия
        name_selectors = [
            'h1::text',
            'h1[itemprop="name"]::text',
            '[data-widget="webProductHeading"] h1::text',
            'meta[property="og:title"]::attr(content)',
        ]

        for selector in name_selectors:
            if selector.startswith('meta'):
                name = response.css(selector).get()
            else:
                name = response.css(selector).get()
            if name and name.strip():
                item['name'] = name.strip()
                self.logger.info(f"📝 Название: {item['name'][:50]}...")
                break

        # Поиск цены
        price_selectors = [
            'span[itemprop="price"]::attr(content)',
            'span[itemprop="price"]::text',
            'span[class*="price"]::text',
            'span[class*="tsHeadline500Medium"]::text',
            'div[data-widget="webPrice"] span::text',
        ]

        for selector in price_selectors:
            price_text = response.css(selector).get()
            if price_text:
                import re
                numbers = re.findall(r'(\d+[\.,]?\d*)', price_text.replace(' ', ''))
                if numbers:
                    price = float(numbers[0].replace(',', '.'))
                    if price > 10000:
                        price = price / 100
                    item['price'] = price
                    self.logger.info(f"💰 Цена: {price}")
                    break

        # Поиск изображения
        img_selectors = [
            'meta[property="og:image"]::attr(content)',
            'img[itemprop="image"]::attr(src)',
            'div[data-widget="webGallery"] img::attr(src)',
        ]

        for selector in img_selectors:
            if selector.startswith('meta'):
                img_url = response.css(selector).get()
            else:
                img_url = response.css(selector).get()
            if img_url:
                if img_url.startswith('//'):
                    img_url = 'https:' + img_url
                item['image_url'] = img_url
                break

        item['url'] = response.url
        item['platform'] = 'ozon'

        if item.get('name') or item.get('price'):
            yield item
        else:

# tracker/parsers/scrapy_spiders/spiders/wildberries_spider.py
import scrapy
import re
import json
from ..items import ProductItem


class WildberriesSpider(scrapy.Spider):
    name = "wildberries"
    allowed_domains = ["wildberries.ru", "www.wildberries.ru", "wb.ru"]

    custom_settings = {
        'DOWNLOAD_DELAY': 1.0,
        'CONCURRENT_REQUESTS': 1,
        'RETRY_TIMES': 3,
        'USER_AGENT': 'Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
    }

    def __init__(self, url=None, *args, **kwargs):
        super().__init__(*args, **kwargs)
        if url:
            self.start_urls = [url]

    def parse(self, response):
        # Проверка на блокировку
        if response.status in [403, 429]:
            self.logger.warning(f"🚫 Блокировка Wildberries (статус {response.status})")
            return

        self.logger.info(f"✅ Страница Wildberries загружена")

        item = ProductItem()

        # 1. Поиск названия
        name_selectors = [
            'h1::text',
            '.product-page__title::text',
            'meta[property="og:title"]::attr(content)',
        ]

        for selector in name_selectors:
            if selector.startswith('meta'):
                name = response.css(selector).get()
            else:
                name = response.css(selector).get()
            if name and name.strip():
                item['name'] = name.strip()
                self.logger.info(f"📝 Название: {item['name'][:50]}...")
                break

        # 2. Поиск цены в JSON
        scripts = response.css('script::text').getall()
        for script in scripts:
            if 'price' in script.lower():
                # Ищем цену в JSON
                match = re.search(r'"price"[^:]*:\s*(\d+)', script)
                if match:
                    price = float(match.group(1)) / 100
                    item['price'] = price
                    self.logger.info(f"💰 Цена: {price}")
                    break

                match = re.search(r'"salePrice"[^:]*:\s*(\d+)', script)
                if match:
                    price = float(match.group(1)) / 100
                    item['price'] = price
                    self.logger.info(f"💰 Цена со скидкой: {price}")
                    break

        # 3. Поиск изображения
        img_selectors = [
            'meta[property="og:image"]::attr(content)',
            '.product-page__img img::attr(src)',
            'img[src*="basket"]::attr(src)',
        ]

        for selector in img_selectors:
            if selector.startswith('meta'):
                img_url = response.css(selector).get()
            else:
                img_url = response.css(selector).get()
            if img_url:
                if img_url.startswith('//'):
                    img_url = 'https:' + img_url
                item['image_url'] = img_url
                break

        item['url'] = response.url
        item['platform'] = 'wildberries'

        if item.get('name') or item.get('price'):
            yield item

from django.db.models.signals import post_migrate
from django.dispatch import receiver
from django_celery_beat.models import PeriodicTask, IntervalSchedule
import json


@receiver(post_migrate)
def setup_periodic_tasks(sender, **kwargs):
    if sender.name == 'tracker':
        # Создаем интервал в 6 часов
        schedule, _ = IntervalSchedule.objects.get_or_create(
            every=6,
            period=IntervalSchedule.HOURS,
        )

        # Задача обновления всех цен
        PeriodicTask.objects.get_or_create(
            interval=schedule,
            name='Update all prices every 6 hours',
            task='tracker.tasks.update_all_prices',
        )

        # Создаем интервал в 24 часа для очистки
        daily_schedule, _ = IntervalSchedule.objects.get_or_create(
            every=24,
            period=IntervalSchedule.HOURS,
        )

        # Задача очистки старой истории

            interval=daily_schedule,

# tracker/tasks.py
from celery import shared_task
from django.core.mail import send_mail
from django.conf import settings
from django.utils import timezone
from .models import Product, PriceHistory, Notification
from .parsers import get_parser
from decimal import Decimal
import logging

logger = logging.getLogger(__name__)


@shared_task
def parse_product_price(product_id):
    """
    Парсинг цены товара с использованием Scrapy и сохранение в историю
    """
    try:
        product = Product.objects.get(id=product_id, is_active=True)
        logger.info(f"🔍 Начинаем Scrapy-парсинг товара {product_id}: {product.url}")

        # Создаем парсер (теперь ScrapyRunner)
        parser = get_parser(product.url)
        if not parser:
            logger.warning(f"❌ Не удалось создать парсер для {product.url}")
            return f"No parser for {product.url}"

        # Парсим данные
        data = parser.parse()

        if not data:
            logger.warning(f"❌ Парсер не вернул данных для {product.url}")

            # Если это первый запуск, повторим через час
            if not product.last_checked:
                parse_product_price.apply_async(args=[product_id], countdown=3600)
                return f"Первая попытка, повторим через час"

            return f"No data from parser for {product.url}"

        # Проверяем наличие цены
        if not data.get('price') or data.get('price') == 0:
            logger.warning(f"⚠️ Цена не найдена для {product.url}. Обновляем только имя/картинку если есть")

            updated = False
            if data.get('name') and (not product.name or product.name == 'Название не найдено'):
                product.name = data['name'][:500]
                updated = True
                logger.info(f"✅ Обновлено имя: {data['name'][:50]}...")

            if data.get('image_url') and not product.image_url:
                product.image_url = data['image_url']
                updated = True
                logger.info(f"✅ Обновлена картинка")

            if updated:
                product.last_checked = timezone.now()
                product.save()
                return f"Обновлены имя/картинка для {product.url}"
            else:
                # Повторим попытку через 6 часов
                parse_product_price.apply_async(args=[product_id], countdown=21600)
                return f"Нет данных для обновления, повторим через 6 часов"

        # Получили цену - обновляем все данные
        old_price = product.current_price
        new_price = Decimal(str(data['price']))

        logger.info(f"💰 Цена найдена: {old_price} -> {new_price}")

        # Обновляем данные товара
        product.current_price = new_price
        product.last_checked = timezone.now()

        if data.get('name') and (not product.name or product.name == 'Название не найдено'):
            product.name = data['name'][:500]
            logger.info(f"✅ Обновлено имя: {data['name'][:50]}...")

        if data.get('image_url') and not product.image_url:
            product.image_url = data['image_url']
            logger.info(f"✅ Обновлена картинка")

        product.save()

        # Сохраняем историю цен
        PriceHistory.objects.create(
            product=product,
            price=new_price
        )
        logger.info(f"📊 История цен сохранена для {product.name}")

        # Проверяем достижение желаемой цены
        if product.desired_price and new_price <= product.desired_price:
            logger.info(f"🎯 Достигнута желаемая цена для {product.name}")
            create_notification.delay(product.id, float(new_price))

        return f"✅ Успешно обновлен {product.name}: {old_price} -> {new_price}"

    except Product.DoesNotExist:
        logger.error(f"❌ Товар {product_id} не найден")
        return f"Product {product_id} not found"
    except Exception as e:
        logger.error(f"❌ Ошибка при парсинге {product_id}: {e}")
        import traceback
        traceback.print_exc()
        return f"Error: {e}"


@shared_task
def create_notification(product_id, price):
    """
    Создание уведомления о достижении цены
    """
    try:
        product = Product.objects.get(id=product_id)

        # Проверяем, не отправляли ли уже уведомление сегодня
        recent_notification = Notification.objects.filter(
            product=product,
            is_sent=True,
            created_at__date=timezone.now().date()
        ).exists()

        if recent_notification:
            logger.info(f"⏭️ Уведомление уже отправлено сегодня для {product.name}")
            return f"Notification already sent today"

        message = f"🎉 Товар '{product.name}' достиг желаемой цены! Текущая цена: {price}₽"

        notification = Notification.objects.create(
            user=product.user,
            product=product,
            message=message
        )
        logger.info(f"📬 Уведомление создано для {product.user.username}")

        # Отправляем email
        send_notification_email.delay(notification.id)

        return f"Notification created for {product.user.username}"
    except Exception as e:
        logger.error(f"Error creating notification: {e}")
        return f"Error: {e}"


@shared_task
def send_notification_email(notification_id):
    """
    Отправка email уведомления
    """
    try:
        notification = Notification.objects.get(id=notification_id, is_sent=False)

        subject = f"💰 Цена достигла желаемого уровня! - {notification.product.name}"
        message = f"""
        Здравствуйте, {notification.user.username}!

        {notification.message}

        Товар: {notification.product.name}
        Текущая цена: {notification.product.current_price}₽
        Желаемая цена: {notification.product.desired_price}₽
        Ссылка на товар: {notification.product.url}

        --
        Price Tracker - сервис отслеживания цен
        """

        # Для разработки используем консольный вывод
        if settings.DEBUG:
            logger.info(f"📧 [DEBUG] Email для {notification.user.email}:")
            logger.info(f"Тема: {subject}")
            logger.info(f"Сообщение: {message[:200]}...")
        else:
            # Для продакшена реальная отправка
            send_mail(
                subject,
                message,
                settings.DEFAULT_FROM_EMAIL or 'noreply@pricetracker.com',
                [notification.user.email],
                fail_silently=False,
            )

        notification.is_sent = True
        notification.sent_at = timezone.now()
        notification.save()

        logger.info(f"✅ Email уведомление отмечено как отправленное для {notification.user.email}")
        return f"Email notification sent to {notification.user.email}"
    except Exception as e:
        logger.error(f"Error sending email: {e}")
        return f"Error sending email: {e}"


@shared_task
def update_all_prices():
    """
    Обновление цен всех активных товаров
    Запускается по расписанию (например, каждые 6 часов)
    """
    try:
        products = Product.objects.filter(is_active=True)
        count = products.count()
        logger.info(f"🚀 Запуск массового обновления цен для {count} товаров")

        for product in products:
            parse_product_price.delay(product.id)
            logger.info(f"⏳ В очередь поставлен товар {product.id}: {product.name}")

        return f"Started parsing {count} products"
    except Exception as e:
        logger.error(f"Error in update_all_prices: {e}")
        return f"Error: {e}"


@shared_task
def cleanup_old_price_history():
    """
    Очистка старой истории цен
    Оставляем только последние 100 записей на товар
    """
    try:
        products = Product.objects.all()
        cleaned = 0

        for product in products:
            history = PriceHistory.objects.filter(product=product).order_by('-created_at')
            if history.count() > 100:
                to_delete = history[100:]
                count = to_delete.count()
                to_delete.delete()
                cleaned += count
                logger.info(f"Очищено {count} старых записей для {product.name}")

        logger.info(f"🧹 Очищено {cleaned} старых записей истории цен")
        return f"Cleaned {cleaned} records"
    except Exception as e:
        logger.error(f"Error in cleanup: {e}")
        return f"Error: {e}"


@shared_task
def retry_failed_parsing():
    """
    Повторная попытка парсинга для товаров без цены
    Запускается раз в час
    """
    try:
        products = Product.objects.filter(
            is_active=True,
            current_price__isnull=True
        ).exclude(
            last_checked__gte=timezone.now() - timezone.timedelta(hours=24)
        )

        count = products.count()
        logger.info(f"🔄 Повторная попытка парсинга для {count} товаров без цены")

        for product in products:
            parse_product_price.delay(product.id)
            logger.info(f"⏳ Повторная попытка для товара {product.id}: {product.name}")

        return f"Retrying {count} products"
    except Exception as e:
        logger.error(f"Error in retry_failed_parsing: {e}")
        return f"Error: {e}"

<script>
function refreshPrice(productId) {
    // Показываем модальное окно
    const modal = new bootstrap.Modal(document.getElementById('refreshModal'));
    const messageEl = document.getElementById('refreshMessage');
    messageEl.textContent = 'Запуск обновления цены...';
    modal.show();

    // Получаем CSRF токен
    const csrftoken = document.querySelector('[name=csrfmiddlewaretoken]').value;

    // Отправляем запрос на обновление
    fetch(`/api/refresh-price/${productId}/`, {
        method: 'POST',
        headers: {
            'X-CSRFToken': csrftoken,
            'Content-Type': 'application/json',
        },
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            messageEl.innerHTML = '<span class="text-success">✅ ' + data.message + '</span>';
            // Перезагружаем страницу через 2 секунды
            setTimeout(() => {
                location.reload();
            }, 2000);
        } else {
            messageEl.innerHTML = '<span class="text-danger">❌ ' + data.message + '</span>';
        }
    })
    .catch(error => {
        messageEl.innerHTML = '<span class="text-danger">❌ Ошибка при обновлении</span>';
        console.error('Error:', error);
    });
}

// Автоматическое обновление статуса для товаров в процессе загрузки

    const loadingElements = document.querySelectorAll('.text-secondary');

            setTimeout(() => {
                location.reload();
            }, 10000);
            break;
        }
    }
});

def index(request):
    """Главная страница"""
    return render(request, 'tracker/index.html')


def register(request):
    """Регистрация нового пользователя"""
    if request.method == 'POST':
        form = CustomUserCreationForm(request.POST)
        if form.is_valid():
            user = form.save()
            login(request, user)
            messages.success(request, '🎉 Регистрация прошла успешно!')
            return redirect('profile')
        else:
            messages.error(request, 'Пожалуйста, исправьте ошибки в форме')
    else:
        form = CustomUserCreationForm()
    return render(request, 'registration/register.html', {'form': form})


@login_required
def profile(request):
    """Профиль пользователя со списком товаров"""
    products = request.user.products.all()
    notifications = request.user.notifications.filter(is_sent=False)[:10]
    return render(request, 'tracker/profile.html', {
        'products': products,
        'notifications': notifications
    })


@login_required
def add_product(request):
    """Добавление нового товара"""
    if request.method == 'POST':
        form = ProductForm(request.POST)
        if form.is_valid():
            product = form.save(commit=False)
            product.user = request.user
            product.save()

            # Запускаем фоновую задачу для парсинга
            parse_product_price.delay(product.id)

            messages.success(request, '✅ Товар успешно добавлен! Цена будет обновлена автоматически.')
            return redirect('profile')
    else:
        form = ProductForm()

    return render(request, 'tracker/add_product.html', {'form': form})


@login_required
def product_detail(request, product_id):
    """Детальная страница товара с графиком цен"""
    product = get_object_or_404(Product, id=product_id, user=request.user)
    price_history = product.price_history.all()[:50]

    dates = [h.created_at.strftime('%d.%m %H:%M') for h in price_history]
    prices = [float(h.price) for h in price_history]

    context = {
        'product': product,
        'price_history': price_history,
        'dates_json': json.dumps(dates[::-1]),
        'prices_json': json.dumps(prices[::-1]),
    }
    return render(request, 'tracker/product_detail.html', context)


@login_required
def delete_product(request, product_id):
    """Удаление товара"""
    product = get_object_or_404(Product, id=product_id, user=request.user)

    if request.method == 'POST':
        product_name = product.name
        product.delete()
        messages.success(request, f'🗑️ Товар "{product_name}" успешно удален')
        logger.info(f"Product {product_id} deleted by user {request.user.username}")
    else:
        messages.error(request, '❌ Неверный метод запроса')

    return redirect('profile')


@login_required
def refresh_price(request, product_id):
    """Ручное обновление цены товара"""
    if request.method == 'POST':
        product = get_object_or_404(Product, id=product_id, user=request.user)
        parse_product_price.delay(product.id)
        messages.success(request, '🔄 Задача на обновление цены запущена')
        return JsonResponse({'success': True})
    return JsonResponse({'success': False}, status=405)


@login_required
def manual_parse_product(request, product_id):
    """Запускает ручной парсинг товара в отдельном окне"""
    product = get_object_or_404(Product, id=product_id, user=request.user)

    # Запускаем скрипт ручного парсинга в фоне
    script_path = os.path.join(os.path.dirname(__file__), '..', 'manual_ozon_parser.py')

    try:
        # Запускаем в новом терминале (для Linux)
        subprocess.Popen([
            'xterm', '-e',
            f'python {script_path} {product.id}; read -p "Нажмите Enter для закрытия..."'
        ])
        messages.success(request,
                         f'🖥️ Запущен ручной парсинг для товара "{product.name}". Решите капчу в открывшемся окне.')
    except FileNotFoundError:
        # Если нет xterm, пробуем gnome-terminal
        try:
            subprocess.Popen([
                'gnome-terminal', '--', 'bash', '-c',
                f'python {script_path} {product.id}; echo "Нажмите Enter для закрытия..."; read'
            ])
            messages.success(request,
                             f'🖥️ Запущен ручной парсинг для товара "{product.name}". Решите капчу в открывшемся окне.')
        except:
            messages.error(request, '❌ Не удалось открыть терминал. Используйте ручной скрипт из командной строки.')

    return redirect('profile')
